create
	int bot counter
		on activation:
			store 5.
		end
		on me > 3:
			store me - 1.
		end
		on me <= 3:
			store 0.
		end
	end
	char bot b
		on activation:
			store 'g'. send.
			store 'u'. send.
			store 'i'. send.
			store 'l'. send.
			store 'l'. send.
			store 'e'.
		end
		on deactivation:
			send.
		end
	end
execute
	activate b, counter.
	deactivate b.
	while counter > 0:
		advance counter.
		activate b.
		deactivate b.
	end
end
